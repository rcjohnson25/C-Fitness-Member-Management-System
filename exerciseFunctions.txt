void initializeExercises(vector<Exercise> &exercises,ifstream &inFile) {
        string line;
        string name;
        double METvalue;
        int count = 0;
        while(getline(inFile, line)) {
                count++;
                if (count % 2 == 1) {
                        name = line;
                }
                else {
                        METvalue = stod(line);
                        Exercise obj(name, METvalue);
                        exercises.push_back(obj);
                }
        }
}

void printListOfExercises(vector<Exercise> &exercises) {
        cout << "List of Exercises:" << endl;
        for (size_t i = 0; i < exercises.size(); i++) {
                exercises[i].printExercise(i+1);
        }
}

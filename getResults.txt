//asks for input to vars, these vars are made into a math equation and then total cals burned is printed
void printResults(int weight, int count, vector<Exercise> &exercises, int &choice) {
    int duration;
    
    cout << "Duration (minutes): ";
    cin >> duration;
    cout << endl;
   
    double MET = exercises[count-1].getMETvalue();
    double caloriesburned = duration * (MET * 3.5 * (weight / 2.205)) / 200;
    
    cout << "The total calories burned: " << static_cast<int>(caloriesburned+.5) << endl;
    cout << endl;
}
